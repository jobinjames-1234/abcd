class InvalidPatientException extends Exception {
    InvalidPatientException(String msg) {
        super(msg);
    }
}

class Patient extends HospitalMember {
    protected int patientId;
    protected String name;
    protected int age;
    protected String disease;

    Patient(int patientId, String name, int age, String disease) throws InvalidPatientException {
        if (age <= 0)
            throw new InvalidPatientException("Invalid Age");

        this.patientId = patientId;
        this.name = name;
        this.age = age;
        this.disease = disease;
    }

    double calculateTreatment(double consultation) throws InvalidPatientException {
        if (consultation < 0)
            throw new InvalidPatientException("Invalid Bill");

        return consultation;
    }

    double calculateTreatment(double consultation, double medicine) throws InvalidPatientException {
        if (consultation < 0 || medicine < 0)
            throw new InvalidPatientException("Invalid Bill");

        return consultation + medicine;
    }

    String prescription() {
        return "Patient: " + name + "\nDisease: " + disease;
    }

    double calculateRoomCharge() {
        return 0;
    }
}

class InPatient extends Patient {
    int days;

    InPatient(int patientId, String name, int age, String disease, int days)
            throws InvalidPatientException {
        super(patientId, name, age, disease);
        this.days = days;
    }

    double calculateRoomCharge() {
        return days * 2000;
    }
}

import java.io.FileWriter;

abstract class LibraryItem {
    abstract void display();
}

class InvalidBookException extends Exception {
    InvalidBookException(String msg) {
        super(msg);
    }
}

class Book extends LibraryItem {
    protected int bookId;
    protected String title;
    protected String author;
    protected boolean available;

    Book(int bookId, String title, String author, boolean available) {
        this.bookId = bookId;
        this.title = title;
        this.author = author;
        this.available = available;
    }

    void search(String title) {
        if (this.title.equalsIgnoreCase(title))
            System.out.println("Book Found: " + this.title);
    }

    void searchByAuthor(String author) {
        if (this.author.equalsIgnoreCase(author))
            System.out.println("Author Found: " + this.author);
    }

    String issueReport() {
        return "Issued Book: " + title;
    }

    void issueBook() throws InvalidBookException {
        if (!available)
            throw new InvalidBookException("Book Not Available");

        available = false;
    }

    void display() {
        System.out.println(title + " by " + author);
    }
}

class PrintedBook extends Book {
    PrintedBook(int id, String title, String author, boolean available) {
        super(id, title, author, available);
    }

    void display() {
        System.out.println("Printed Book: " + title);
    }
}

import java.io.*;

abstract class LibraryItem {

    protected int bookId;
    protected String title;

    public LibraryItem(int bookId, String title) {
        this.bookId = bookId;
        this.title = title;
    }

    abstract void display();
}

class InvalidBookOperationException extends Exception {

    public InvalidBookOperationException(String message) {
        super(message);
    }
}

class Book extends LibraryItem {

    protected String author;
    protected boolean available;

    static Book[] books = new Book[100];
    static int count = 0;

    public Book(int bookId, String title,
                String author, boolean available) {

        super(bookId, title);

        this.author = author;
        this.available = available;
    }

    public void searchBook(String title) {

        boolean found = false;

        for (int i = 0; i < count; i++) {

            if (books[i].title.equalsIgnoreCase(title)) {

                books[i].display();
                found = true;
            }
        }

        if (!found) {
            System.out.println("Book Not Found!");
        }
    }

    public void searchBookByAuthor(String author) {

        boolean found = false;

        for (int i = 0; i < count; i++) {

            if (books[i].author.equalsIgnoreCase(author)) {

                books[i].display();
                found = true;
            }
        }

        if (!found) {
            System.out.println("Author Not Found!");
        }
    }

    public String generateIssueReport() {

        String report = "\n===== ISSUE REPORT =====\n";
        report += "Book ID : " + bookId + "\n";
        report += "Title   : " + title.toUpperCase() + "\n";
        report += "Author  : " + author + "\n";
        report += "Status  : Issued\n";
        report += "========================";

        return report;
    }

    public void issueBook()
            throws InvalidBookOperationException {

        if (!available) {

            throw new InvalidBookOperationException(
                    "Book Already Issued!");
        }

        available = false;

        saveToFile();
    }

    public void returnBook()
            throws InvalidBookOperationException {

        if (available) {

            throw new InvalidBookOperationException(
                    "Book Already Available!");
        }

        available = true;
    }

    public void saveToFile() {

        try {

            FileWriter fw =
                    new FileWriter(
                            "issued_books.txt", true);

            fw.write("Book ID : " + bookId + "\n");
            fw.write("Title   : " + title + "\n");
            fw.write("Author  : " + author + "\n");
            fw.write("------------------------\n");

            fw.close();

            System.out.println(
                    "Issued Book Saved to File!");

        } catch (IOException e) {

            System.out.println("File Error!");
        }
    }

    @Override
    void display() {

        System.out.println("\nBook ID : " + bookId);
        System.out.println("Title   : " + title);
        System.out.println("Author  : " + author);
        System.out.println("Status  : "
                + (available ? "Available"
                : "Issued"));
    }

    static class LibraryRules {

        public static void showRules() {

            System.out.println(
                    "\n===== LIBRARY RULES =====");

            System.out.println(
                    "1. Return books within 7 days");

            System.out.println(
                    "2. Keep books clean");

            System.out.println(
                    "3. Fine for late return");

            System.out.println(
                    "=========================");
        }
    }
}

class FoodOrder extends Order {

    protected String foodItem;
    protected int quantity;
    protected double price;

    static FoodOrder[] orders = new FoodOrder[100];
    static int count = 0;

    public FoodOrder(int orderId,
                     String customerName,
                     String foodItem,
                     int quantity,
                     double price)
            throws InvalidQuantityException {

        super(orderId, customerName);

        if (quantity <= 0) {

            throw new InvalidQuantityException(
                    "Invalid Quantity!");
        }

        this.foodItem = foodItem;
        this.quantity = quantity;
        this.price = price;
    }

    public double calculateBill(double deliveryCharge) {

        return (quantity * price)
                + deliveryCharge;
    }

    public double calculateBill(double deliveryCharge,
                                double expressCharge) {

        return (quantity * price)
                + deliveryCharge
                + expressCharge;
    }

    @Override
    double calculateBill() {

        return quantity * price;
    }

    public String generateInvoice(double amount) {

        String invoice =
                "\n======= CUSTOMER INVOICE =======\n";

        invoice += "Order ID      : "
                + orderId + "\n";

        invoice += "Customer Name : "
                + customerName.toUpperCase()
                + "\n";

        invoice += "Food Item     : "
                + foodItem.toLowerCase()
                + "\n";

        invoice += "Quantity      : "
                + quantity + "\n";

        invoice += "Total Amount  : "
                + amount + "\n";

        invoice += "================================";

        return invoice;
    }

    public void saveToFile(double amount) {

        try {

            FileWriter fw =
                    new FileWriter(
                            "orders.txt", true);

            fw.write("Order ID : "
                    + orderId + "\n");

            fw.write("Customer : "
                    + customerName + "\n");

            fw.write("Food Item: "
                    + foodItem + "\n");

            fw.write("Quantity : "
                    + quantity + "\n");

            fw.write("Bill     : "
                    + amount + "\n");

            fw.write("--------------------------\n");

            fw.close();

            System.out.println(
                    "Order Saved to File!");

        } catch (IOException e) {

            System.out.println("File Error!");
        }
    }

    public void display() {

        System.out.println("\nOrder ID      : "
                + orderId);

        System.out.println("Customer Name : "
                + customerName);

        System.out.println("Food Item     : "
                + foodItem);

        System.out.println("Quantity      : "
                + quantity);

        System.out.println("Price         : "
                + price);
    }
}
