import java.io.FileWriter;
import java.io.IOException;

// Abstract Class
abstract class Person {
    abstract void displayInfo();
}

// User Defined Exception
class InvalidMarkException extends Exception {
    InvalidMarkException(String msg) {
        super(msg);
    }
}

// Student Class
class Student extends Person {

    public int studentId;
    protected String name;
    private String department;
    int marks;

    // Constructors
    Student() {
        studentId = 0;
        name = "Unknown";
        department = "None";
        marks = 0;
    }

    Student(int id, String n, String d, int m)
            throws InvalidMarkException {

        if (m < 0 || m > 100)
            throw new InvalidMarkException("Invalid Marks");

        studentId = id;
        name = n;
        department = d;
        marks = m;
    }

    // Method Overloading
    int calculateTotal(int a, int b, int c) {
        return a + b + c;
    }

    int calculateTotal(int a, int b,
                       int c, int d, int e) {
        return a + b + c + d + e;
    }

    void displayStudent() {
        String report =
                "\nID : " + studentId +
                "\nName : " + name +
                "\nDepartment : " + department +
                "\nMarks : " + marks;

        System.out.println(report);
    }

    // File Handling
    void saveReport() {

        try {

            FileWriter fw =
                    new FileWriter("student.txt", true);

            fw.write("ID : " + studentId + "\n");
            fw.write("Name : " + name + "\n");
            fw.write("Marks : " + marks + "\n\n");

            fw.close();

            System.out.println("Report Saved");

        } catch (IOException e) {
            System.out.println(e);
        }
    }

    void displayInfo() {
        displayStudent();
    }
}

// Separate Result Class
class Result {

    void generateResult(int marks) {

        if (marks >= 50)
            System.out.println("PASS");
        else
            System.out.println("FAIL");

        if (marks >= 90)
            System.out.println("Grade A");
        else if (marks >= 75)
            System.out.println("Grade B");
        else
            System.out.println("Grade C");
    }
}

// Inheritance
class MCAStudent extends Student {

    MCAStudent(int id, String n,
               String d, int m)
            throws InvalidMarkException {

        super(id, n, d, m);
    }

    // Method Overriding
    @Override
    void displayStudent() {

        System.out.println("MCA Student Details");
        super.displayStudent();
    }
}

// Multithreading
class ReportThread extends Thread {
    public void run() {
        System.out.println("Generating Report...");
    }
}

class SaveThread extends Thread {

    Student s;

    SaveThread(Student s) {
        this.s = s;
    }

    public void run() {
        s.saveReport();
    }

// Main Class
public class StudentManagement {

    public static void main(String[] args) {

        try {

            MCAStudent s1 =
                    new MCAStudent(
                            101,
                            "Rahul",
                            "MCA",
                            85);

            Student arr[] = new Student[2];
            arr[0] = s1;

            s1.displayStudent();

            System.out.println(
                    "3 Subject Total : "

            System.out.println(
                    "5 Subject Total : "

                                    80, 70, 90, 88, 95));

            Result r = new Result();
            r.generateResult(s1.marks);

            ReportThread t1 = new ReportThread();
            SaveThread t2 = new SaveThread(s1);

import java.io.FileWriter;
import java.io.IOException;

abstract class BankUser {
    abstract void userType();
}

class InsufficientBalanceException extends Exception {

    InsufficientBalanceException(String msg) {
        super(msg);
    }
}

class Account extends BankUser {

    int accNo;
    String customerName;
    double balance;

    Account() {
        accNo = 0;
        customerName = "Unknown";
        balance = 0;
    }

    Account(int a, String n, double b) {
        accNo = a;
        customerName = n;
        balance = b;
    }

    // Method Overloading
    void deposit(int amount) {
        balance += amount;
    }

    void deposit(double amount) {
        balance += amount;
    }

    void withdraw(double amount)
            throws InsufficientBalanceException {

        if (amount > balance)
            throw new InsufficientBalanceException(
                    "Insufficient Balance");

        balance -= amount;
    }

    void display() {

        String summary =
                "\nAccount No : " + accNo +
                "\nCustomer : " + customerName +
                "\nBalance : " + balance;

        System.out.println(summary);
    }

    void saveTransaction() {

        try {

            FileWriter fw =
                    new FileWriter("transaction.txt", true);

            fw.write(customerName +
                    " " + balance + "\n");

            fw.close();

        } catch (IOException e) {
            System.out.println(e);
        }
    }

    void userType() {
        System.out.println("Bank User");
    }
}

class SavingsAccount extends Account {

    SavingsAccount(int a, String n, double b) {
        super(a, n, b);
    }

    @Override
    void withdraw(double amount)
            throws InsufficientBalanceException {

        if (balance - amount < 500)
            throw new InsufficientBalanceException(
                    "Minimum Balance Required");

        balance -= amount;
    }
}

class CurrentAccount extends Account {

    CurrentAccount(int a, String n, double b) {
        super(a, n, b);
    }

    @Override
    void withdraw(double amount)
            throws InsufficientBalanceException {

        balance -= amount;
    }
}

class BalanceThread extends Thread {

    Account a;

    BalanceThread(Account a) {
        this.a = a;
    }

    public void run() {
        System.out.println("Checking Balance...");
    }
}

class TransactionThread extends Thread {

    public void run() {
        System.out.println("Processing Transaction...");
    }
}

public class BankingApp {

    public static void main(String[] args) {

        try {

            SavingsAccount s =
                    new SavingsAccount(
                            1001,
                            "Anu",
                            5000);

            Account arr[] = new Account[2];
            arr[0] = s;

            s.deposit(500);
            s.deposit(250.5);

            s.withdraw(1000);

            s.display();

            s.saveTransaction();

            BalanceThread t1 =
                    new BalanceThread(s);

            TransactionThread t2 =
                    new TransactionThread();

            t1.start();
            t2.start();

        } catch (Exception e) {
            System.out.println(e.getMessage());
        }
    }
}

// Book Class
class Book extends LibraryItem {

    int bookId;
    String title;
    String author;
    boolean available;

    // Constructor
    Book(int id, String t,
         String a, boolean av) {

        bookId = id;
        title = t;
        author = a;
        available = av;
    }

    // Method Overloading

        System.out.println("Searching Book by Title");

        System.out.println("Searching Book by Author");

            throws InvalidBookException {

            throw new InvalidBookException(
                    "Book Not Available");

        available = false;

        System.out.println("Book Issued");
    }

    void display() {

        String details =
                "\nBook ID : " + bookId +
                "\nTitle : " + title +
                "\nAuthor : " + author;

        System.out.println(details);

    // File Handling

        try {

            FileWriter fw =
                    new FileWriter("library.txt", true);

            System.out.println("Book Saved");

    abstract double calculateBill();
}

    InvalidQuantityException(String msg) {
        super(msg);
    }
}

// FoodOrder Class
class FoodOrder extends Order {

    int orderId;
    String customerName;
    String foodItem;
    int quantity;

    // Constructor
    FoodOrder(int id,
              String c,
              String item,
              int q)
            throws InvalidQuantityException {

        if (q <= 0)
            throw new InvalidQuantityException(
                    "Invalid Quantity");

        orderId = id;
        customerName = c;
        foodItem = item;
        quantity = q;
    }

    // Method Overloading
    double calculateBill(int price) {
        return quantity * price;
    }

    double calculateBill(int price,
                         int expressCharge) {
        return quantity * price + expressCharge;
    }

    void display() {

        String invoice =
                "\nOrder ID : " + orderId +
                "\nCustomer : " + customerName +
                "\nFood Item : " + foodItem +
                "\nQuantity : " + quantity;

        System.out.println(invoice);
    }

    // File Handling
    void saveOrder() {

        try {

            FileWriter fw =
                    new FileWriter("orders.txt", true);

            fw.write(customerName +
                    " " + foodItem + "\n");

            fw.close();

            System.out.println("Order Saved");

        } catch (IOException e) {
            System.out.println(e);
        }
    }

    double calculateBill() {
        return 0;
    }
}
